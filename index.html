} catch(e) { console.error('loadAccounts:', e); } } async function writeCC(dt, descricao, valor, tipo, ref_id, banco) { try { const body = {dt, descricao, valor: parseFloat(valor), tipo}; if (ref_id) body.ref_id = ref_id;
